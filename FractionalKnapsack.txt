import java.util.*;

public class FractionalKnapsack {

    // Item structure
    static class Item {
        int weight;
        int profit;
        double ratio;

        Item(int weight, int profit) {
            this.weight = weight;
            this.profit = profit;
            this.ratio = (double) profit / weight; // profit/weight ratio
        }
    }

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // Input number of items
        System.out.print("Enter number of items: ");
        int n = sc.nextInt();

        Item[] items = new Item[n];

        // Input profit and weight of each item
        System.out.println("Enter weight and profit of each item:");
        for (int i = 0; i < n; i++) {
            int w = sc.nextInt();
            int p = sc.nextInt();
            items[i] = new Item(w, p);
        }

        // Input capacity of knapsack
        System.out.print("Enter capacity of knapsack: ");
        int capacity = sc.nextInt();

        // Sort items based on ratio (profit/weight) in descending order
        Arrays.sort(items, (a, b) -> Double.compare(b.ratio, a.ratio));

        double totalProfit = 0;

        for (Item item : items) {
            if (capacity == 0) break;

            if (item.weight <= capacity) {
                // Take the whole item
                capacity -= item.weight;
                totalProfit += item.profit;
            } else {
                // Take fraction of item
                totalProfit += item.ratio * capacity;
                capacity = 0; // knapsack full
            }
        }

        System.out.println("Maximum Profit = " + totalProfit);
    }
}
